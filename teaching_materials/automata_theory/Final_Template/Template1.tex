\textbf{ حتما پیش از پاسخ‌دادن به نکات زیر توجه کنید:}\\
\textcolor{black}{ توجه کنید که برای گرفتن نمره‌ی کامل از پرسش‌های ۱ تا ۳، باید برای ادعاهایتان دلیل کافی بیاورید. با این‌حال، با توجه به محدودیت زمان، از شما انتظار نمی‌رود که اثبات‌هایی مفصل و با جزئیات کامل ارائه کنید. سعی کنید نکات مهم را به‌طور مختصر خاطرنشان کنید و کلیت طرح اثبات را بیان کنید. در مواردی که برای پاسخ به پرسشی یک الگوریتم، ماشین محاسبه، گرامر، عبارت منظم و ... ارائه می‌کنید، نیاز به اثبات درستی آن‌چه ارائه کرده‌اید نیست؛ اما سعی کنید با توضیحی مختصر، شهود پشت طراحیتان را به مصحح منتقل کنید.\\
	در پاسخ به سوال ۴، برخلاف سه پرسش قبل، از شما انتظار می‌رود به‌طور دقیق ادعاهایتان را اثبات کنید. با این همه، اگر نیازی به طراحی ماشین تورینگ داشتید، چه در این سوال و چه در پرسش‌های قبل آن، می‌توانید از توصیف سطح بالا (الگوریتمی) ماشین تورینگ استفاده کنید و نیازی به طراحی سطح پایین (طراحی ساختار ماشین و تابع انتقال آن) نیست!}

\problem{درست و نادرست}
درستی یا نادرستی هر یک از گزاره‌های زیر را با ذکر اجمالی دلیل، تعیین کنید.\\
\begin{enumerate}
\item 
 اگر در گراف انتقال حالت متناظر با یک $ DFA$، جهت یال‌ها را برعکس کنیم، گراف حاصل نمایش یک $DFA$ است.
 \item
 اگر در گراف انتقال حالت متناظر با یک $ DFA$، جهت یال‌ها را برعکس کنیم، گراف حاصل نمایش یک $NFA$ است.
\item 
 یک زبان منظم، نامتناهی است اگر و فقط اگر گراف انتقال حالت متناظر با $ DFA$ای که آن را می‌پذیرد دارای دور باشد. 
\item 
مساله‌ی متناهی بودن یک زبان منظم، تصمیم‌پذیر است.
\item 
روی الفبای تک‌عضوی 
$\Sigma=\{a\}$، زبان‌هایی وجود دارند که تورینگ-تشخیص‌پذیر نیستند.
\item 
یک ماشین محاسبه‌ی غیرقطعی، ماشینی است که قطعی نیست.
\item 
اگر یک $PDA$ برای زبان 
$L$
داشته باشیم، در این صورت با اجرای رشته‌ی دلخواه 
$w$
روی ماشین، می‌توان تعیین کرد که این رشته عضو 
$L$
است یا عضو 
$L^c$.
\item 
با داشتن توصیف یک 
$PDA$، 
تعیین عضویت رشته‌ی دلخواه 
$w$
در 
$L$، 
تصمیم‌ناپذیر است.
\end{enumerate}

\problem{کلاس ناشناخته} 
به دلخواه خود یکی از دو پرسش زیر را انتخاب و به آن پاسخ دهید.\\
\begin{enumerate}
	\item 
 الفبای دلخواه 
	$\Sigma$
	را در نظر بگیرید. مدلی از ماشین‌های محاسبه ارائه کنید که کلاس همه‌ی زبان‌هایی (روی الفبای 	$\Sigma$) که توسط ماشین‌هایی از این مدل، \underline{تشخیص‌ داده می‌شوند}، در سلسله مراتب زبانی چامسکی جایگاهی مانند \textit{کلاس ناشناخته}‌ی شکل زیر داشته باشند:\\
\begin{figure}[h]
	\includegraphics[width=10cm]{1.png}
	\centering
\end{figure}\\
توجه کنید که کلاس ناشناخته‌ی مدل محاسبه‌ی شما باید شامل تعداد نامتناهی زبان غیرمنظم باشد و هم‌چنین باید تعداد نامتناهی زبان متناهی وجود داشته باشند که در این کلاس نیستند.
	\item 
	
	الفبای دلخواه 
	$\Sigma$
	را در نظر بگیرید. مدلی از ماشین‌های محاسبه ارائه کنید که کلاس همه‌ی زبان‌هایی (روی الفبای 	$\Sigma$) که توسط ماشین‌هایی از این مدل، \underline{تشخیص‌ داده می‌شوند}، در سلسله مراتب زبانی چامسکی جایگاهی مانند \textit{کلاس ناشناخته}‌ی شکل زیر داشته باشند:\\
	\begin{figure}[h]
		\includegraphics[width=10cm]{2.png}
		\centering
	\end{figure}
	توجه کنید که کلاس ناشناخته‌ی مدل محاسبه‌ی شما باید شامل تعداد نامتناهی زبان 
	\lr{R.E.}
	 باشد که تصمیم‌پذیر نیستند و هم‌چنین باید تعداد نامتناهی زبان تصمیم‌پذیر وجود داشته باشند که در این کلاس نیستند.

\end{enumerate}

\problem{دسته‌بندی زبان‌ها}
زبان‌هایی که در ادامه می‌آیند را در یکی از دسته‌های 

\begin{itemize}
	\item منظم
	\item مستقل از متن ولی غیرمنظم
	\item تصمیم‌پذیر ولی غیر مستقل از متن
	\item تشخیص‌پذیر ولی تصمیم‌ناپذیر
	\item هیچ‌کدام
\end{itemize}
قرار دهید و برای ادعایتان دلایل کافی بیاورید.

\begin{enumerate}
	\item $L_1=\{1^x**1^y=1^z|x,y,z\in \mathbb{N}~,~z=x^y\}$
	\item 
	$L_2=\{ a^ib^j c^k|i,j,k\geq 0~,~ j\neq i~ \text{یا}~ j\neq k \}$
	\item 
	$L_3=\{w\in \{0,1\}^*| \text{  $w$ کدینگ یک ماشین تورینگ است } \}$
\end{enumerate}
\problem{مسائل کامل کلاس 
\lr{R.E.}
و 
\lr{R}
}
فرض کنید 
$\mathcal{C}$
کلاسی از زبان‌ها باشد. زبان 
$A$
را 
$-\mathcal{C}$کامل 
گوییم هرگاه هر دو شرط زیر برقرار باشند:
\begin{enumerate}
	\item 
	برای هر زبان 
	$L\in \mathcal{C}$، 
	$$L\leq_m A.$$
	\item 
	$A\in \mathcal{C}$.
\end{enumerate}
با توجه به تعریف بالا ثابت کنید:
\begin{enumerate}
	\item 
	هر زبان نابدیهی تصمیم‌پذیر، 
	$-\mathcal{R}$کامل 
	است. (زبان 
	$L$
	را نابدیهی می‌گویند هرگاه 
	$L\neq \emptyset , \Sigma^*$.)
	\item 
	زبان 
	$A_{TM}$، 
	$-R.E.$کامل است.
\end{enumerate}
